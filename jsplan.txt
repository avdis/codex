						// hitting escape
						// clicking away
						$(document)
							.off('keyup.purchase-order')
							.on('keyup.purchase-order', function(event) {
								if (event.which == keycode.ESC) {
									$drop.remove();
								};
							})
							.off('mouseup.purchase-order')
							.on('mouseup.purchase-order', function(event) {
								if (! $(event.target).closest('.js-item').length) {
									$drop.remove();
								};
							});
